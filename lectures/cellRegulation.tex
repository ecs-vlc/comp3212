\begin{slide}
\section{The First Law of Protein Sequences}

\begin{PauseHighLight}
  \begin{center}
    \emph{Similar structures perform similar functions}\pause
  \end{center}
  \begin{itemize}
  \item We can learn about the function of a protein by comparing with
    known proteins\pause
  \item Because of evolution, many proteins are common to a huge number
    of organism\pause
  \item Caveat: the converse is not necessarily true; dissimilar
    structures can have a similar function (convergent
    evolution)\pause
  \end{itemize}
\end{PauseHighLight}

\end{slide}

\begin{slide}
\section{Gene Regulation Networks}

\begin{PauseHighLight}
  \begin{itemize}
  \item Proteins can regulate their own expression\pause
  \item Proteins can regulate many other proteins\pause
  \item Protein $A$ can regulate protein $B$ which in turn regulates
    protein $A$ (if these are inhibitory then you get a flip-flop)\pause
  \item In general the regulation forms a complex circuit\pause
  \end{itemize}
\end{PauseHighLight}

\end{slide}
